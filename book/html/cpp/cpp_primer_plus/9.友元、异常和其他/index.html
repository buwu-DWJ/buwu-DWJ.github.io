    public:

    };

    QueueTP(int qs = Q_SIZE);
    ~QueueTP();
    bool isempty() const

        return items == 0;
    }
    bool isfull() const
    {
        return items == qsize;
    }

    {

    }

// QueueTP methods
template <class Item>
QueueTP<Item>::QueueTP(int qs) : qsize(qs)
{
    front = rear = 0;
    items = 0;
}

QueueTP<Item>::~QueueTP()
{
    Node * temp;
    while (front != 0)      // while queue is not yet empty
    {
        temp = front;
        front = front->next;
        delete temp;

{
    if (isfull())

    else

}
